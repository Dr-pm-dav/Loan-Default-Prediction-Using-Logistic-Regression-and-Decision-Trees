from sklearn.tree import DecisionTreeClassifier
from sklearn.metrics import classification_report, accuracy_score, ConfusionMatrixDisplay
import matplotlib.pyplot as plt

# Initialize and train a Decision Tree classifier
tree_model = DecisionTreeClassifier(random_state=42)
tree_model.fit(X_train, y_train)

# Predict on test data
tree_predictions = tree_model.predict(X_test)

# Evaluate performance
tree_accuracy = accuracy_score(y_test, tree_predictions)
tree_report = classification_report(y_test, tree_predictions)

# Display results
print(f"Decision Tree Accuracy: {tree_accuracy:.4f}")
print("Classification Report:\n", tree_report)

# Plot confusion matrix
ConfusionMatrixDisplay.from_estimator(tree_model, X_test, y_test)
plt.title('Decision Tree Confusion Matrix')
plt.show()

from sklearn.model_selection import GridSearchCV

# Set up parameter grid for hyperparameter tuning
param_grid = {
    'C': [0.01, 0.1, 1, 10, 100],  # Regularization strength
    'solver': ['lbfgs', 'liblinear']  # Common solvers for logistic regression
}

# Initialize the model
log_reg = LogisticRegression(max_iter=500, random_state=42)

# GridSearchCV for hyperparameter tuning
grid_search = GridSearchCV(estimator=log_reg, param_grid=param_grid, cv=5, scoring='accuracy', n_jobs=-1)
grid_search.fit(X_train, y_train)

# Best parameters and model evaluation
best_log_reg = grid_search.best_estimator_
best_params = grid_search.best_params_
best_accuracy = grid_search.best_score_

# Evaluate best model on the test set
y_pred_best = best_log_reg.predict(X_test)
final_accuracy = accuracy_score(y_test, y_pred_best)
final_report = classification_report(y_test, y_pred_best)

print(f"Best Hyperparameters: {best_params}")
print(f"Cross-validated Training Accuracy: {best_accuracy:.4f}")
print(f"Test Set Accuracy: {final_accuracy:.4f}")
print("Final Classification Report:\n", final_report)

# Plot Confusion Matrix for best model
ConfusionMatrixDisplay.from_estimator(best_log_reg, X_test, y_test)
plt.title('Best Logistic Regression Confusion Matrix (Tuned)')
plt.show()
